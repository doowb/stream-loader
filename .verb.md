## Usage

```js

var dest = require('dest');

var src = loader();
src('*.js')
  .pipe(jshint())
  .pipe(minify())
  .pipe(dest('dist/'));
```

## with vinyl

convert stream-loader files into vinyl files

```js
var File = require('vinyl');

var src = loader(function (err, file, cb) {
  return cb(null, new File(file));
});

src('*.js')
  .pipe(jshint())
  .pipe(minify())
  .pipe(dest('dist/'));
```
